\usepackage{amsmath,amssymb}

\title{Quadratic Residue}
\author{}


\begin{document}
	
	\maketitle
	
	Denote $\mathbb{F}_p=\left\{0,\dots, p-1\right\}$.
	
	\begin{enumerate}
		\item[\textbf{P1 (O0)}] 
		Show that 
		\[
		x^2 + ax \equiv 0 \pmod{p}
		\] 
		has at most 2 solutions for $x\in \mathbb{F}_p$ for a prime \(p\) and integer \(a\).
		
		\item[\textbf{P2 (O1)}] 
		Show that 
		\[
		x^2 + a \equiv 0 \pmod{p}
		\] 
		has at most 2 solutions for $x\in \mathbb{F}_p$ for a prime \(p\) and integer \(a\).
		
		\item[\textbf{P3 (O2)}] 
		Show that 
		\[
		x^2 + ax + b = 0 \pmod{p}
		\] 
		has at most 2 solutions for $x\in \mathbb{F}_p$ for a prime $p$ and integers \(a,b\).
		
		\item[\textbf{P4 (O2)}] 
		Find all integer solutions to 
		\[
		x^2 = 2y^{10} + 11.
		\]
		
		\item[\textbf{P5 (O2)}] 
		Let \(p\) be a prime. Show that if there exists \(x\) such that 
		\[
		x^2 \equiv a \pmod{p}
		\] 
		and there does not exist \(y\) such that 
		\[
		y^2 \equiv b \pmod{p},
		\] 
		then there does not exist \(z\) such that 
		\[
		z^2 \equiv ab \pmod{p}.
		\]
	\end{enumerate}
