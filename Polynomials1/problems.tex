	\section*{Polynomials}
	
	$\mathbb{Z}[x]$ is the set of integer coefficient polynomials.
	
	A polynomial $P \in \mathbb{Z}[x]$ is \emph{irreducible} over $\mathbb{Z}$ if there do not exist non-constant polynomials $Q, R \in \mathbb{Z}[x]$ such that
	\[
	P(x) = Q(x)R(x).
	\]
	
	\subsection*{Problems}
	
	\begin{itemize}
		\item \textbf{P1 (O1)} Show that $x^2 + 1$ is irreducible over $\mathbb{Z}$.
		
		\item \textbf{P2 (O1)} Find all primes $p$ such that $x^p + 1$ is irreducible.
		
		\item \textbf{P3 (O1)} For which integers $a$ is $x^2 + ax + 1$ irreducible over $\mathbb{Q}$?
		
		\item \textbf{P4 (O1)} Let $P \in \mathbb{Z}[x]$ be a polynomial that is not irreducible. Show that $P(n)$ is prime for only finitely many $n \in \mathbb{Z}$.
		
		\item \textbf{P5 (MAOL).} Does there exist a polynomial $P \in \mathbb{Z}[x]$ that does not have integer roots, such that for every prime $p$, there exists $n \in \mathbb{Z}$ with
		\[
		p \mid P(n)\,?
		\]
	\end{itemize}
