	\title{Diophantine Equation Problem Set}
	
	\maketitle
	
	\begin{enumerate}
		\item[\textbf{P1} (O1)] Find all integers \(x,y \in \mathbb{Z}\) such that
		\[
		x^{3} - y^{3} = 67.
		\]
		
		\item[\textbf{P2} (O1)] Find all integers \(x,y \in \mathbb{Z}\) and all natural numbers \(n \in \mathbb{N}\) such that
		\[
		x + x^{2} + x^{3} = 67^{n}.
		\]
		
		\item[\textbf{P3} (O1)] Find all integers \(x,y \in \mathbb{Z}\) and all natural numbers \(n \in \mathbb{N}\) such that
		\[
		x + y + xy = 67^{n} - 1.
		\]
		
		\item[\textbf{P4} (O1)] Find all natural numbers \(x \in \mathbb{N}\) and primes \(p\) such that the number
		\[
		x^{p} - x
		\]
		has exactly two prime divisors.
		
		\item[\textbf{P5} (O2)] Show that for \(x > 1\), the number
		\[
		1 + x^{2} + x^{4}
		\]
		cannot be a prime.
	\end{enumerate}
